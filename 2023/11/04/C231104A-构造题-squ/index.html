题解代码

#include <bits/stdc++.h>
using namespace std;
const int N=303;
int a[N][N];
int main(){
    freopen("squ.in","r",stdin);
    freopen("squ.out","w",stdout);
    int n;
    cin>>n;
    if(n%2==1){
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                a[i][j]=(i+j)%n;
            }
        }
    }else{
        for(int i=0;i<n-1;i++){
            for(int j=0;j<n-1;j++){
                a[i][j]=(i+j)%(n-1);
            }
        }
        a[n-1][n-1]=n-1;
        for(int i=0;i<n-1;i++){
            int j=(i+n-2)%(n-1);
            a[n-1][j]=a[i][n-1]=a[i][j];
            a[i][j]=n-1;
        }
    }
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            printf("%d ",a[i][j]+1);
        }
        puts("");
    }
}
