void dijkstra(int root){
	q.push({0,root}),d[root]=0;
	while(q.size()){
		int x=q.top().second;q.pop();
		for(auto p:ft[x]){
			int y=p.first,w=p.second;
			if(d[x]+w<d[y]){
				d[y]=d[x]+w;
				q.push({-d[y],y});
			}
		}
	}
}

#include <bits/stdc++.h>
#define int long long
#define N 200010
using namespace std;
int n, m;
int u, v, c;
int first[N * 3], nxt[N * 5], to[N * 5], w[N * 5], et;
void add(int u, int v, int c) {
  nxt[++et] = first[u];
  first[u] = et;
  to[et] = v;
  w[et] = c;
}
int dis[N][4];
#define pii pair<int, int>
#define piii pair<int, pii>
#define fi first
#define se second
#define mp make_pair
#define inf 0x3f3f3f3f
bool vis[N][4];
void dij() {
  memset(dis, 0x3f, sizeof(dis));
  priority_queue<piii, vector<piii>, greater<piii>> q;
  dis[1][0] = 0;
  q.emplace(0, mp(1, 0));
  while (!q.empty()) {
    int u = q.top().se.fi;
    int s = q.top().se.se;
    q.pop();
    if (vis[u][s]) continue;
    vis[u][s] = 1;
    for (int i = first[u]; i; i = nxt[i]) {
      if (!(s & 1)) {//0-->1  2-->3 (Ух░min)
        if (dis[to[i]][s | 1] > dis[u][s] + w[i] * 2) {
          dis[to[i]][s | 1] = dis[u][s] + w[i] * 2;
          q.emplace(dis[to[i]][s | 1], mp(to[i], s | 1));
        }
      }
      if (!(s & 2)) {//0-->2   1-->3  (Ух░max)
        if (dis[to[i]][s | 2] > dis[u][s]) {
          dis[to[i]][s | 2] = dis[u][s];
          q.emplace(dis[to[i]][s | 2], mp(to[i], s | 2));
        }
      }
      if (!s) {//0-->3 (Ух░ТЎ«жђџУЙ╣тѕ░3)
        if (dis[to[i]][3] > dis[u][s] + w[i]) {
          dis[to[i]][3] = dis[u][s] + w[i];
        } 
      }
      if (dis[to[i]][s] > dis[u][s] + w[i]) {//0-->0  1-->1  2-->2  3-->3(тюетйЊтЅЇт▒ѓУх░ТЎ«жђџУЙ╣)
        dis[to[i]][s] = dis[u][s] + w[i];
        q.emplace(dis[to[i]][s], mp(to[i], s));
      }
    }
  }
}
