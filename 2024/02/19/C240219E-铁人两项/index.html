$\qquad$еЕґдЄ≠жЬЙдЄАдЄ™ж†СзЪДжГЕеЖµгАВеѓєдЇОдЄАж£µж†СпЉМжИСдїђеПѓдї•еЕИз°ЃеЃЪдЄАдЄ™еЫЇеЃЪзЪДзВєпЉМзДґеРОдїОињЩдЄ™зВєеЉАеІЛ $dfs$ пЉМеЬ®еНКиЈѓдЄКиЃ°зЃЧдїОжЬАеЉАеІЛзЪДзВє $x$ пЉМиµ∞еИ∞ељУеЙНзВє $y$ дЄ≠зЪДиЈѓеЊДжЭ°жХ∞пЉМжЧґйЧіе§НжЭВеЇ¶ $O(n^2)$ гАВжНҐдЄАдЄ™жАЭиЈѓпЉМе¶ВжЮЬдї•жЮЪдЄЊдЄ≠йЧізЪДзВєпЉИиЃЊдЄЇ $x$ пЉЙпЉМйВ£дєИжИСдїђеПѓдї•йАЪињЗеЬ®ж±ВеЕґе≠Рж†СзЪДе§Іе∞П $siz[x]$ зЪДжЧґеАЩиЃ°зЃЧеЗЇзїПињЗељУеЙНзВєзЪДиЈѓеЊДжЭ°жХ∞гАВ

$\qquad$еѓєдЇОж£ЃжЮЧдЄ≠зЪДжѓПдЄАж£µж†СпЉМжИСдїђеПѓдї•е∞ЖжѓПдЄАдЄ™зВєеПМињЮйАЪеИЖйЗПзЬЛдљЬдЄАдЄ™жХідљУпЉМеИЩз≠Фж°ИзЪДиЃ°зЃЧеПШжИРдЇЖж±В $ans+=2\cdot siz[x]\cdot siz[y]\cdot w[x]$ пЉИ$w[x]$ и°®з§ЇзЪДжШѓељУеЙНзВєеПМзЪДзВєжХ∞пЉЙпЉМзФ±дЇОжѓПдЄАдЄ™зВєеПМињЮйАЪеИЖйЗПиЃ°зЃЧзВєзЪДдЄ™жХ∞зЪДжЧґеАЩйГљеМЕеРЂдЇЖеЃГзЪДеЙ≤зВєпЉМеЫ†ж≠§жИСдїђеЬ®иЃ°зЃЧиі°зМЃзЪДжЧґеАЩдЉЪжККеЙ≤зВєе§ЪиЃ°зЃЧдЄАжђ°пЉМеЫ†ж≠§жИСдїђйЬАи¶БиАГиЩСе¶ВдљХеОїйЩ§ињЩдЄ™ељ±еУНпЉМзФ±дЇОжЬАзїИиЃ°зЃЧеЗЇжЭ•зЪДиЈѓеЊДйГљдЉЪдєШдЄКељУеЙНзВєеПМдЄ≠зВєзЪДдЄ™жХ∞пЉМжЙАдї•жИСдїђеПѓдї•зЫіжО•йАЙжЛ©е∞ЖжЙАжЬЙзЪДеЬЖзВєзЪД $w[x]$ иµЛеАЉдЄЇ $-1$ пЉМзДґеРОе∞ЖжЙАжЬЙжЦєзВєеѓєеЇФзЪДзВєзЪДдЄ™жХ∞иµЛеАЉдЄЇељУеЙНзВєеПМзЪДзВєзЪДдЄ™жХ∞гАВ

#include<bits/stdc++.h>
#include<vector>
#include<stack>
#include<set>
#define For(i,l,r) for(int i=l;i<=r;i++) 
#define Rof(i,l,r) for(int i=l;i>=r;i--)
using namespace std;
#define P pair<int,int>
#define int long long
inline int input(){int x;return cin>>x,x;}
const int inf = 0x3f3f3f3f3f3f3f3f;
const int N = 201234;

vector<int> ft[N];
stack<int> st;

int n,m,ans;
int w[N];

int dfn[N],low[N],t;
int scc_cnt,num;
vector<int> tr[N];
void tarjan(int x){
	dfn[x]=low[x]=++t,num++;
	st.emplace(x),w[x]=-1;
	for(auto y:ft[x]){
		if(!dfn[y]){
			tarjan(y) , low[x] = min(low[x],low[y]);
			if(dfn[x] == low[y]){
				scc_cnt++;
				while(st.top() != y){
					tr[st.top()].emplace_back(scc_cnt);
					tr[scc_cnt].emplace_back(st.top());
					w[scc_cnt]++;st.pop();
				}st.pop();w[scc_cnt]+=2;
				tr[y].emplace_back(scc_cnt),tr[scc_cnt].emplace_back(y);
				tr[x].emplace_back(scc_cnt),tr[scc_cnt].emplace_back(x);
			}
		}else low[x] = min(low[x],dfn[y]);
	}
}

int siz[N];
void dfs_ans(int x,int fa){
	siz[x] = x <= n;
	for(auto y:tr[x]){
		if(y == fa) continue;
		dfs_ans(y,x);
		ans += 2 * siz[x] * siz[y] * w[x];
		siz[x] += siz[y];
	}ans += 2 * siz[x] * (num - siz[x]) * w[x];
}

signed main(){
	ios::sync_with_stdio(false),cin.tie(0),cout.tie(0);
	cin>>n>>m;For(i,1,m){
		int x=input(),y=input();
		ft[x].emplace_back(y);
		ft[y].emplace_back(x);
	}int Tim = clock();
	
	scc_cnt=n;
	
	For(i,1,n) if(!dfn[i]) tarjan(i),dfs_ans(i,0),num=0;
	
	cout<<ans<<'\n';
	return cerr<<"TIME:"<<(clock()-Tim)/1024.,0;
}
